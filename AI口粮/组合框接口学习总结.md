# 神途游戏客户端接口学习总结 - 组合框控件接口

## 概述

组合框控件是神途游戏客户端中结合了文本输入和下拉列表功能的高级UI组件，它允许用户通过编辑框输入内容或从下拉列表中选择预定义选项。组合框广泛应用于需要用户进行选择或输入的界面，如角色创建、设置选项、任务选择等。

### 核心功能

- 创建组合框控件
- 管理组合框的条目（添加、清空）
- 设置和获取选中条目
- 配置组合框的显示属性（编辑框大小、下拉列表高度）
- 设置组合框的图片资源（背景图、滚动条图）
- 响应组合框的交互事件

### 应用场景

- 角色创建时的职业选择
- 游戏设置中的音效和画质选项
- 任务系统中的任务选择
- 商店中的物品分类选择
- 角色属性分配时的属性类型选择

## 基础操作

### 1. 创建组合框控件

```lua
-- 创建组合框控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local _GUIHandle = GUI:ComboBoxCreate(_Parent, "JobSelectComboBox", "职业选择", 100, 100, 200, 30)

if _GUIHandle ~= 0 then
    dbg("组合框控件创建成功")

    -- 设置编辑框大小
    GUI:ComboBoxSetEditSize(_GUIHandle, 150, 30)

    -- 设置下拉背景高度

    -- 添加选项
    GUI:ComboBoxAddString(_GUIHandle, "战士")
    GUI:ComboBoxAddString(_GUIHandle, "法师")
    GUI:ComboBoxAddString(_GUIHandle, "道士")

    -- 设置默认选中项
    GUI:ComboBoxSetCurSelect(_GUIHandle, 0)

    dbg("组合框控件创建失败")
end
```

### 2. 组合框事件处理

```lua
-- 组合框事件处理
local function RegisterComboBoxEvents()
    local _GUIHandle = GetWindowHandle("JobSelectComboBox")

    if _GUIHandle ~= 0 then
        -- 注册选中项改变事件
        GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_comboBox_selectChange, "OnComboBoxSelectChange")

        -- 注册编辑框点击事件
        GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_mouse_lbClick, "OnComboBoxEditClick")

        -- 注册编辑框文本改变事件
        local _EditHandle = GUI:ComboBoxGetEdit(_GUIHandle)
        if _EditHandle ~= 0 then
            GUI:WndRegistScript(_EditHandle, RDWndBaseCL_edit_change, "OnComboBoxEditChange")
        end
    end
end

-- 选中项改变事件处理
function OnComboBoxSelectChange(_GUIHandle)
    local _SelectIndex = GUI:ComboBoxGetCurSelect(_GUIHandle)
    local _SelectText = ""

    -- 获取列表框句柄
    local _ListHandle = GUI:ComboBoxGetListHandle(_GUIHandle)

    if _ListHandle ~= 0 then
        _SelectText = GUI:ListBoxGetLineString(_ListHandle, _SelectIndex)
    end

    dbg(string.format("选中项改变: 索引=%d, 文本=%s", _SelectIndex, _SelectText))

    -- 执行相应操作
    OnJobSelectChanged(_SelectIndex)
end

-- 编辑框点击事件处理
function OnComboBoxEditClick(_GUIHandle)
    dbg("编辑框被点击")

    -- 显示提示信息
    ShowSystemMessage("请输入或选择职业")
end

-- 编辑框文本改变事件处理
function OnComboBoxEditChange(_GUIHandle)
    local _Text = GUI:EditGetTextM(_GUIHandle)
    dbg(string.format("编辑框文本改变: %s", _Text))

    -- 验证输入内容
    ValidateJobInput(_Text)
end
```

```lua
        -- 读取当前音效设置并设置选中项
        local _CurrentVolume = GetCurrentSoundVolume() -- 0-3

        -- 设置滚动条图片
        GUI:ComboBoxSetScrollBarImage(_GUIHandle, 1853900004)

        -- 注册选中项改变事件
        GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_comboBox_selectChange, "OnSoundSettingsChanged")

        dbg("音效设置组合框创建成功")
    end
end

-- 音效设置改变事件
function OnSoundSettingsChanged(_GUIHandle)
    local _SelectIndex = GUI:ComboBoxGetCurSelect(_GUIHandle)

    -- 根据选中项设置音效音量
    local VolumeLevels = {100, 70, 40, 0}

    if VolumeLevels[_SelectIndex] then
        SetSoundVolume(VolumeLevels[_SelectIndex])

        -- 更新系统通知
        local VolumeText = {
            "高", "中", "低", "关闭"
        }

        ShowSystemMessage(string.format("音效设置已更改为：%s", VolumeText[_SelectIndex]))

        -- 保存设置
        SaveSettingsToFile("SoundVolume", VolumeLevels[_SelectIndex])
    end
end
```

```lua
-- 组合框编辑框输入验证
local function ValidateJobInput(_Text)
    local ValidJobs = {"战士", "法师", "道士", "弓箭手", "刺客"}

    -- 检查输入是否为有效的职业名称
    local _IsValidJob = false

    for _, _Job in ipairs(ValidJobs) do
        if string.lower(_Text) == string.lower(_Job) then
            _IsValidJob = true
            break
        end
    end

    if _IsValidJob then
        -- 输入有效，显示确认提示
        ShowValidationMessage("输入有效的职业", "success")
    else
        -- 输入无效，显示警告
        ShowValidationMessage("请输入有效的职业名称", "warning")
    end
end

-- 显示验证消息
function ShowValidationMessage(_Text, _Type)
    local _MessageHandle = CreateMessageWindow()

    if _MessageHandle ~= 0 then
        -- 设置消息类型（成功/警告/错误）
        GUI:WndSetVisibleM(_MessageHandle, true)
        GUI:WndSetTextM(_MessageHandle, _Text)

        -- 设置消息窗口颜色
        if _Type == "success" then
            GUI:WndSetBackgroundColorM(_MessageHandle, 0x00FF00) -- 绿色
        elseif _Type == "warning" then
            GUI:WndSetBackgroundColorM(_MessageHandle, 0xFFFF00) -- 黄色
        else
            GUI:WndSetBackgroundColorM(_MessageHandle, 0xFF0000) -- 红色
        end

        -- 自动隐藏消息窗口
        GUI:WndAddTimer(_MessageHandle, 1000, 2000, 1, "OnMessageTimer")
    end
end
```
