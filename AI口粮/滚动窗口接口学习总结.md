# 神途游戏客户端接口学习总结 - 滚动窗口控件接口

## 概述

滚动窗口控件是神途游戏客户端中用于处理分页内容显示的重要UI组件，它允许用户通过左右滑动来查看不同页面的内容。每个页面包含完整的内容，滑动时会切换到相邻页面。滚动窗口广泛应用于游戏中的背包、仓库、技能书等需要大量物品展示的界面。

### 核心功能

- 创建滚动窗口控件
- 设置滚动窗口的可视区域、页大小和滚动速度
- 获取和设置当前页索引
- 管理滚动窗口的子控件
- 响应滚动和滑动事件

### 应用场景

- 背包物品展示
- 角色仓库界面
- 技能书展示
- 商店物品浏览
- 任务奖励展示

## 基础操作

### 1. 创建滚动窗口控件

```lua
-- 创建滚动窗口控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local PageWidth = 410 -- 背包单页的宽度
local PageCount = 4 -- 背包总页数
local OnePageCount = 25 -- 每页背包格子的数量

local _PageView = GUI:WndViewCreat(_Parent, "PakageView") -- 创建滚动窗口控件

if _PageView ~= 0 then
    -- 设置自定义数据，用于页面计数
    GUI:WndSetUserData(_PageView, PageCount, 0)

    -- 设置穿透类型为WndView的滚动
    GUI:WndSetPenetratType(_PageView, 2)

    -- 设置使用背景缓存标记
    GUI:WndSetFlagsM(_PageView, flg_wndBase_useBkBuffer)

    -- 设置滚动窗口的大小
    GUI:WndSetSizeM(_PageView, PageWidth * PageCount, 430)

    -- 设置滚动窗口的位置
    GUI:WndSetPosM(_PageView, 20, 57)

    -- 设置滚动窗口的可视区域
    GUI:WndSetViewPortSize(_PageView, 403, 430)

    -- 设置滚动窗口的页大小
    GUI:WndViewSetPageSize(_PageView, PageWidth, 430)

    -- 设置滚动窗口的滚动速度（数值越大，滚动速度越快，默认值为10）
    GUI:WndViewSetAutoScrollRatio(_PageView, 30)

    dbg("滚动窗口创建成功")
else
    dbg("滚动窗口创建失败")
end
```

```lua
-- 动态添加物品框作为子控件
local function AddItemSlots()
    local _PageView = GetWindowHandle("PakageView")
    if _PageView ~= 0 then
        local PageWidth = 410 -- 单页宽度
        local PageCount = 4 -- 总页数
        local OnePageCount = 25 -- 每页物品数量
        local intervalX = 82
        local intervalY = 80

        for PageIndex = 0, PageCount - 1 do -- 创建物品框作为子控件，用于物品的展示和交互
            local LineIndex = 1
            local posY = 4
            for i = 1, OnePageCount do
                local posX = intervalX * (LineIndex - 1) + PageWidth * PageIndex + 4
                local _ItemSlot = GUI:ItemCtrlCreate(_PageView, "Item"..(i - 1 + OnePageCount * PageIndex), 1853800010, posX, posY, 66, 66)

                if _ItemSlot ~= 0 then
                    GUI:WndRegistScript(_ItemSlot, RDWndBaseCL_mouse_lbUp, "Package.onPackageItem")
                end

                if LineIndex % 5 == 0 then
                    LineIndex = 1
                    posY = posY + intervalY
                else
                    LineIndex = LineIndex + 1
                end
            end
        end

        dbg("物品框子控件创建完成")
    end
end
```

```lua
-- 滚动事件处理
local function OnScrollChanged()
    local _PageView = GetWindowHandle("PakageView")
    local _CurrentPage = GUI:WndViewGetCurrentPage(_PageView)

    dbg("滚动到页面: " .. (_CurrentPage + 1))

    -- 根据当前页面更新UI状态
    UpdatePageIndicator(_CurrentPage)
end

-- 更新页面指示器
local function UpdatePageIndicator(_CurrentPage)
    local PageCount = 4
    local PageIndicator = GetWindowHandle("PageIndicator")

    if PageIndicator ~= 0 then
        for i = 0, PageCount - 1 do
            local _Indicator = GUI:WndGetChildByName(PageIndicator, "Page"..(i + 1))
            if _Indicator ~= 0 then
                if i == _CurrentPage then
                    -- 当前页面指示器高亮
                    GUI:WndSetVisibleM(_Indicator, true)
                    GUI:WndSetEnableM(_Indicator, false)
                else
                    -- 其他页面指示器正常状态
                    GUI:WndSetVisibleM(_Indicator, true)
                    GUI:WndSetEnableM(_Indicator, true)
                end
            end
        end
    end
end

-- 页面点击事件处理
local function OnPageClick(_Handle)
    local PageIndex = tonumber(string.match(GUI:WndGetIDM(_Handle), "Page(%d)")) - 1
    local _PageView = GetWindowHandle("PakageView")

    if _PageView ~= 0 and PageIndex >= 0 and PageIndex < 4 then
        -- 跳转到指定页面
        GUI:WndViewSetCurrentPage(_PageView, PageIndex + 1)
    end
end
```

### 3. 响应滑动手势

```lua
-- 滑动手势识别
local _StartX = 0
local _IsTouching = false

-- 触摸开始事件
function OnTouchBegin(_X, _Y)
    _StartX = _X
    _IsTouching = true
end

-- 触摸结束事件
function OnTouchEnd(_X, _Y)
    if not _IsTouching then
        return
    end

    _IsTouching = false

    local _DeltaX = _X - _StartX
    local _PageView = GetWindowHandle("PakageView")

    if _PageView ~= 0 then
        local _CurrentPage = GUI:WndViewGetCurrentPage(_PageView)

        -- 滑动距离足够触发翻页
        if math.abs(_DeltaX) > 50 then
            local _NextPage = _CurrentPage

            if _DeltaX < 0 then
                -- 向左滑动，显示下一页
                if _CurrentPage < 3 then
                    _NextPage = _CurrentPage + 1
                end
            else
                -- 向右滑动，显示上一页
                if _CurrentPage > 0 then
                    _NextPage = _CurrentPage - 1
                end
            end

            if _NextPage ~= _CurrentPage then
                GUI:WndViewSetCurrentPage(_PageView, _NextPage + 1)
            end
        end
    end
end
```

```lua
function AddPageIndicator()
    local _Parent = GetWindowHandle("MainWindow")
    local _IndicatorPanel = GUI:WndCreateWnd(_Parent, "PageIndicator", 200, 490, 150, 30)

    if _IndicatorPanel ~= 0 then
        local PageCount = 4
        local Interval = 30

        for i = 0, PageCount - 1 do
            local _PageDot = GUI:ImageCreate(_IndicatorPanel, "Page"..(i + 1), i * Interval, 0, 10, 10)

            if _PageDot ~= 0 then
                if i == 0 then
                    -- 第一页指示器高亮
                    GUI:ImageSetImage(_PageDot, 1900000047)
                else
                    -- 其他页面指示器正常状态
                    GUI:ImageSetImage(_PageDot, 1900000048)
                end

                GUI:WndRegistScript(_PageDot, RDWndBaseCL_mouse_lbUp, "OnPageClick")
            end
        end
    end
end

-- 加载背包数据
function LoadBackpackData()
    local PlayerData = GetPlayerData()
    local _PageView = GetWindowHandle("BackpackView")

    if PlayerData.Backpack and _PageView ~= 0 then
        local PageWidth = 410
        local intervalX = 82
        local intervalY = 80

        for i, Item in ipairs(PlayerData.Backpack) do
            local PageIndex = math.floor(i / 25)
            local LineIndex = math.floor((i % 25) / 5) + 1
            local posX = intervalX * (LineIndex - 1) + PageWidth * PageIndex + 4
            local posY = intervalY * math.floor((i % 25) / 5) + 4

            local _ItemSlot = GUI:WndGetChildByName(_PageView, "Item"..i)

            if _ItemSlot ~= 0 then
                SetItemToSlot(_ItemSlot, Item)
            end
        end

        dbg("背包数据加载完成")
    end
end
```

```lua
    local _PageView = GetWindowHandle("BackpackView")

            GUI:WndSetVisibleM(_ItemSlot, true)

    local _PageView = GetWindowHandle("BackpackView")
    local PageCount = 4

        if PageIndex ~= _CurrentPage then
            for i = PageIndex * OnePageCount + 1, (PageIndex + 1) * OnePageCount do
                local _ItemSlot = GUI:WndGetChildByName(_PageView, "Item"..i)

                if _ItemSlot ~= 0 then
                    GUI:WndSetVisibleM(_ItemSlot, false)
```

```lua
-- 物品显示缓存机制
local _ItemCache = {}

function SetItemToSlot(_Slot, _ItemData)
    local _CacheKey = _ItemData.ID .. "_" .. _ItemData.Count

    -- 检查缓存
    if _ItemCache[_CacheKey] then
        -- 使用缓存的显示属性
        GUI:ItemCtrlSetItemID(_Slot, _ItemCache[_CacheKey].ID)
        GUI:ItemCtrlSetItemCount(_Slot, _ItemCache[_CacheKey].Count)
        GUI:ItemCtrlSetItemBind(_Slot, _ItemCache[_CacheKey].IsBound)
    else
        -- 设置物品属性
        GUI:ItemCtrlSetItemID(_Slot, _ItemData.ID)
        GUI:ItemCtrlSetItemCount(_Slot, _ItemData.Count)
        GUI:ItemCtrlSetItemBind(_Slot, _ItemData.IsBound)

        -- 缓存物品显示属性
        _ItemCache[_CacheKey] = {
            ID = _ItemData.ID,
            Count = _ItemData.Count,
            IsBound = _ItemData.IsBound
        }
    end
end
```

```lua
local function OptimizeScrolling()
    local _PageView = GetWindowHandle("BackpackView")

    if _PageView ~= 0 then
        -- 减少滚动窗口子控件数量

        SimplifyUIElements()
    end

    dbg("滚动窗口性能优化完成")

-- 减少可见区域物品框数量
function ReduceVisibleItems()
    local _PageView = GetWindowHandle("BackpackView")
    local PageCount = 4
    local OnePageCount = 25

    for PageIndex = 0, PageCount - 1 do
        for i = 1, OnePageCount do
            local _ItemSlot = GUI:WndGetChildByName(_PageView, "Item"..(i - 1 + OnePageCount * PageIndex))

            if _ItemSlot ~= 0 then
                -- 只显示当前页的物品框
                if PageIndex ~= GetCurrentPageIndex() then
                    GUI:WndSetVisibleM(_ItemSlot, false)
                end
            end
        end
    end
end

-- 优化背景缓存
function OptimizeBackgroundCache()
    local _PageView = GetWindowHandle("BackpackView")

        GUI:WndSetPenetratType(_PageView, 2)

    end
end
```
