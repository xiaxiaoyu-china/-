# 神途游戏客户端接口学习总结 - 列表框控件接口

## 概述

列表框控件是神途游戏客户端中用于显示和选择选项的重要UI组件，它允许用户在一个列表中选择一个或多个选项。列表框支持多种配置选项，包括列数设置、行高调整、背景图片自定义以及滚动条样式修改等。

### 核心功能

- **基础功能**：
  - 创建列表框控件
  - 添加、删除和清空列表项
  - 设置和获取选中项
  - 支持多列显示

- **高级功能**：
  - 自定义列表框的背景和滚动条图片
  - 设置列表项的显示格式和列宽
  - 控制列表框的滑动和选中行为
  - 获取列表框的各种属性信息

### 应用场景

- **角色选择**：在创建角色时选择职业或种族
- **装备管理**：在背包中显示和选择装备
- **技能树**：显示和选择可学习的技能
- **任务系统**：显示和选择可接受的任务
- **设置选项**：在游戏设置中选择音效、画质等选项

## 基础操作

### 1. 创建列表框控件

```lua
-- 创建列表框控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local _GUIHandle = GUI:ListBoxCreate(_Parent, "ExampleListBox", 100, 100, 200, 200)

if _GUIHandle ~= 0 then
    dbg("列表框创建成功")

    -- 设置列表框属性
    GUI:ListBoxSetColumnNum(_GUIHandle, 1) -- 设置为单列表框
    GUI:ListBoxSetOffsetY(_GUIHandle, 25) -- 设置行高为25像素
else
    dbg("列表框创建失败")
end
```

### 2. 添加和删除列表项

```lua
-- 添加列表项
local _GUIHandle = GetWindowHandle("ExampleListBox")

if _GUIHandle ~= 0 then
    -- 添加普通文本项
    GUI:ListBoxAddStr(_GUIHandle, "选项1")
    GUI:ListBoxAddStr(_GUIHandle, "选项2")
    GUI:ListBoxAddStr(_GUIHandle, "选项3")

    -- 添加自定义颜色的文本项
    GUI:ListBoxAddString(_GUIHandle, "红色选项", COLOR_RED)
    GUI:ListBoxAddString(_GUIHandle, "蓝色选项", COLOR_BLUE)
    GUI:ListBoxAddString(_GUIHandle, "绿色选项", COLOR_GREEN)
else
    dbg("列表框句柄无效")
end

-- 删除列表项
local function DeleteListItem(_Index)
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        GUI:ListBoxDelString(_GUIHandle, _Index)
        dbg(string.format("删除第%d项成功", _Index))
    end
end

-- 清空列表框
local function ClearListBox()
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        GUI:ListBoxClear(_GUIHandle)
        dbg("列表框清空成功")
    end
end
```

### 3. 设置和获取选中项

```lua
-- 设置选中项
local function SetSelectedItem(_LineIndex, _ColumnIndex)
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        GUI:ListBoxSetCurSel(_GUIHandle, _LineIndex, _ColumnIndex)
        dbg(string.format("选中第%d行第%d列", _LineIndex, _ColumnIndex))
    end
end

-- 获取选中项
local function GetSelectedItem()
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        local _LineIndex = GUI:ListBoxGetCurSelLine(_GUIHandle)
        local _ColumnIndex = GUI:ListBoxGetCurSelCol(_GUIHandle)

        if _LineIndex >= 0 and _ColumnIndex >= 0 then
            local _ItemText = GUI:ListBoxGetString(_GUIHandle, _LineIndex, _ColumnIndex)
            dbg(string.format("选中项：第%d行第%d列，内容：%s", _LineIndex, _ColumnIndex, _ItemText))
        else
            dbg("未选中任何项")
        end
    end
end
```

### 2. 自定义样式

```lua
-- 自定义列表框样式
local function CustomizeListBox()
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        -- 设置默认颜色
        GUI:ListBoxSetDefaultColor(_GUIHandle, RDLB_COLOR_BK, COLOR_WHITE) -- 背景颜色
        GUI:ListBoxSetDefaultColor(_GUIHandle, RDLB_COLOR_TEXT, COLOR_BLACK) -- 文本颜色
        GUI:ListBoxSetDefaultColor(_GUIHandle, RDLB_COLOR_SEL_BK, COLOR_BLUE) -- 选中项背景颜色
        GUI:ListBoxSetDefaultColor(_GUIHandle, RDLB_COLOR_SEL_TEXT, COLOR_WHITE) -- 选中项文本颜色

        -- 设置滚动条图片
        GUI:ListBoxSetScrollBarImage(_GUIHandle,
            1900000001, -- 垂直滚动条背景图片
            1900000002, -- 垂直滚动条滑块图片
            1900000003, -- 垂直滚动条上按钮图片
            1900000004, -- 垂直滚动条下按钮图片
            0, -- 水平滚动条背景图片（不使用）
            0, -- 水平滚动条滑块图片（不使用）
            0, -- 水平滚动条左按钮图片（不使用）
            0 -- 水平滚动条右按钮图片（不使用）
        )

        -- 设置条目填充方式
        GUI:ListBoxSetFillAllLineBack(_GUIHandle, true) -- 整行填充背景
    end
end
```

        end

### 1. 列表项动态加载

-- 动态加载列表项以提高性能
local function LoadItemsInBatches(_StartIndex, _EndIndex)
    local _GUIHandle = GetWindowHandle("ExampleListBox")

    if _GUIHandle ~= 0 then
        -- 清空现有项
        GUI:ListBoxClear(_GUIHandle)

        -- 加载指定范围内的项
        for i = _StartIndex, _EndIndex do
            local _ItemText = string.format("动态加载项 %d", i)
            GUI:ListBoxAddStr(_GUIHandle, _ItemText)
        end

        dbg(string.format("已加载项范围：%d - %d", _StartIndex, _EndIndex))
    end
end

-- 分页加载示例
local _CurrentPage = 0
local _ItemsPerPage = 10

local function LoadNextPage()
    local _StartIndex = _CurrentPage * _ItemsPerPage
    local _EndIndex = (_CurrentPage + 1) * _ItemsPerPage - 1

    LoadItemsInBatches(_StartIndex, _EndIndex)
    _CurrentPage = _CurrentPage + 1
end
