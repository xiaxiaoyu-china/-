# 神途游戏客户端接口学习总结 - 复选框控件接口

## 概述

复选框控件是神途游戏客户端中用于提供选择功能的基础UI组件，允许用户在一组选项中选择一个或多个选项。它具有简单直观的界面，支持文本标签和选中/未选中状态的可视化反馈。

### 核心功能

- 创建复选框控件
- 设置和获取选中状态
- 配置文本内容和字体样式
- 控制选中状态的可改变性
- 调整图片缩放和文本对齐方式

### 应用场景

- 游戏设置界面中的选项选择
- 角色创建时的属性配置
- 装备管理中的筛选条件
- 技能学习时的选择界面
- 任务系统中的选项确认

## 基础操作

### 1. 创建复选框控件

```lua
-- 创建复选框控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local _GUIHandle = GUI:CheckBoxCreate(_Parent, "ExampleCheckBox", 1900112013, "是否开启音效", 100, 100)

if _GUIHandle ~= 0 then
    dbg("复选框控件创建成功")
    -- 设置默认选中状态
    GUI:CheckBoxSetCheck(_GUIHandle, true)
else
    dbg("复选框控件创建失败")
end
```

### 2. 设置和获取选中状态

```lua
-- 设置复选框选中状态
local _GUIHandle = GetWindowHandle("ExampleCheckBox")
if _GUIHandle ~= 0 then
    GUI:CheckBoxSetCheck(_GUIHandle, true) -- 选中
    -- GUI:CheckBoxSetCheck(_GUIHandle, false) -- 未选中
end

-- 获取复选框选中状态
local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)
dbg("复选框选中状态: " .. tostring(_IsChecked))
```

### 3. 配置文本内容和字体

```lua
-- 设置文本内容
GUI:CheckBoxSetText(_GUIHandle, "是否开启音效")

-- 设置字体
GUI:CheckBoxSetFontM(_GUIHandle, "SIMLI18") -- 设置为宋体，18号字

-- 获取字体信息
local _FontName = GUI:CheckBoxGetFontM(_GUIHandle)
dbg("当前字体: " .. _FontName)
```

## 事件处理

### 1. 复选框状态改变事件

```lua
-- 注册复选框状态改变事件
local _GUIHandle = GetWindowHandle("ExampleCheckBox")
GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_checkbox_change, "OnCheckBoxStateChanged")

-- 事件处理函数
function OnCheckBoxStateChanged(_GUIHandle)
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)
    local _Text = GUI:CheckBoxGetText(_GUIHandle)

    dbg(string.format("%s 状态改变为: %s", _Text, tostring(_IsChecked)))

    -- 根据选中状态执行相应操作
    if _IsChecked then
        EnableSoundEffect()
    else
        DisableSoundEffect()
    end
end
```

### 2. 点击事件处理

```lua
-- 注册鼠标点击事件
local _GUIHandle = GetWindowHandle("ExampleCheckBox")
GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_mouse_lbClick, "OnCheckBoxClick")

-- 点击事件处理函数
function OnCheckBoxClick(_GUIHandle)
    dbg("复选框被点击")

    -- 手动切换选中状态
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)
    GUI:CheckBoxSetCheck(_GUIHandle, not _IsChecked)
end
```

## 实际应用示例

### 1. 游戏设置界面

```lua
-- 创建游戏设置界面的复选框组
local function CreateGameSettings()
    local _Parent = LuaGlobal["AttachPartent"]

    -- 创建音效设置复选框
    local _SoundCheckbox = GUI:CheckBoxCreate(_Parent, "SoundCheckbox", 1900112013, "开启音效", 100, 100)
    if _SoundCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_SoundCheckbox, GetSoundEnabled())
        GUI:WndRegistScript(_SoundCheckbox, RDWndBaseCL_checkbox_change, "OnSoundSettingChanged")
    end

    -- 创建音乐设置复选框
    local _MusicCheckbox = GUI:CheckBoxCreate(_Parent, "MusicCheckbox", 1900112013, "开启音乐", 100, 130)
    if _MusicCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_MusicCheckbox, GetMusicEnabled())
        GUI:WndRegistScript(_MusicCheckbox, RDWndBaseCL_checkbox_change, "OnMusicSettingChanged")
    end

    -- 创建震动设置复选框
    local _VibrationCheckbox = GUI:CheckBoxCreate(_Parent, "VibrationCheckbox", 1900112013, "开启震动", 100, 160)
    if _VibrationCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_VibrationCheckbox, GetVibrationEnabled())
        GUI:WndRegistScript(_VibrationCheckbox, RDWndBaseCL_checkbox_change, "OnVibrationSettingChanged")
    end
end

-- 音效设置改变事件
function OnSoundSettingChanged(_GUIHandle)
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)
    SetSoundEnabled(_IsChecked)
    dbg(string.format("音效设置: %s", tostring(_IsChecked)))
end

-- 音乐设置改变事件
function OnMusicSettingChanged(_GUIHandle)
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)

    dbg(string.format("音乐设置: %s", tostring(_IsChecked)))

-- 初始化设置界面
CreateGameSettings()
```

### 2. 角色创建属性选择

```lua
-- 创建角色属性选择界面
local function CreateCharacterAttributes()
    local _Parent = LuaGlobal["AttachPartent"]

    -- 力量属性
    local _StrengthCheckbox = GUI:CheckBoxCreate(_Parent, "StrengthCheckbox", 1900112013, "力量", 100, 100)
    if _StrengthCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_StrengthCheckbox, false)
        GUI:WndRegistScript(_StrengthCheckbox, RDWndBaseCL_checkbox_change, "OnAttributeChanged")
    end

    -- 敏捷属性
    local _AgilityCheckbox = GUI:CheckBoxCreate(_Parent, "AgilityCheckbox", 1900112013, "敏捷", 100, 130)
    if _AgilityCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_AgilityCheckbox, false)
        GUI:WndRegistScript(_AgilityCheckbox, RDWndBaseCL_checkbox_change, "OnAttributeChanged")
    end

    -- 智力属性
    local _IntelligenceCheckbox = GUI:CheckBoxCreate(_Parent, "IntelligenceCheckbox", 1900112013, "智力", 100, 160)
    if _IntelligenceCheckbox ~= 0 then
        GUI:CheckBoxSetCheck(_IntelligenceCheckbox, false)
        GUI:WndRegistScript(_IntelligenceCheckbox, RDWndBaseCL_checkbox_change, "OnAttributeChanged")
    end
end

-- 属性选择事件
function OnAttributeChanged(_GUIHandle)
    local _WndId = GUI:WndGetIDM(_GUIHandle)
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)
    local _AttributeName = ""

    -- 根据控件ID判断属性类型
    if _WndId == "StrengthCheckbox" then
        _AttributeName = "力量"
        SetCharacterAttribute("Strength", _IsChecked)
    elseif _WndId == "AgilityCheckbox" then
        _AttributeName = "敏捷"
        SetCharacterAttribute("Agility", _IsChecked)
    elseif _WndId == "IntelligenceCheckbox" then
        _AttributeName = "智力"
        SetCharacterAttribute("Intelligence", _IsChecked)
    end

    dbg(string.format("%s 属性选择: %s", _AttributeName, tostring(_IsChecked)))
end

-- 初始化属性选择界面
CreateCharacterAttributes()
```

## 优化与性能考虑

### 1. 批量创建优化

```lua
-- 批量创建复选框控件
local function CreateCheckboxGroup(_Parent, _CheckboxData)
    for i, data in ipairs(_CheckboxData) do
        local _GUIHandle = GUI:CheckBoxCreate(_Parent, data.id, 1900112013, data.text, data.x, data.y)

        if _GUIHandle ~= 0 then
            -- 设置默认选中状态
            if data.checked ~= nil then
                GUI:CheckBoxSetCheck(_GUIHandle, data.checked)
            end

            -- 注册事件
            if data.eventHandler then
                GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_checkbox_change, data.eventHandler)
            end
        end
    end
end

-- 复选框数据配置
local _CheckboxData = {
    { id = "Checkbox1", text = "选项1", x = 100, y = 100, checked = true, eventHandler = "OnCheckbox1Changed" },
    { id = "Checkbox2", text = "选项2", x = 100, y = 130, checked = false, eventHandler = "OnCheckbox2Changed" },
    { id = "Checkbox3", text = "选项3", x = 100, y = 160, checked = true, eventHandler = "OnCheckbox3Changed" }
}

-- 使用批量创建函数
CreateCheckboxGroup(LuaGlobal["AttachPartent"], _CheckboxData)
```

### 2. 事件处理优化

```lua
-- 统一事件处理函数
function OnCheckboxChanged(_GUIHandle)
    local _WndId = GUI:WndGetIDM(_GUIHandle)
    local _IsChecked = GUI:CheckBoxGetCheck(_GUIHandle)

    -- 通用处理逻辑
    dbg(string.format("%s 状态改变为: %s", _WndId, tostring(_IsChecked)))

    -- 特定处理逻辑
    if _WndId == "SoundCheckbox" then
        SetSoundEnabled(_IsChecked)
    elseif _WndId == "MusicCheckbox" then
        SetMusicEnabled(_IsChecked)
    elseif _WndId == "VibrationCheckbox" then
        SetVibrationEnabled(_IsChecked)
    end
end

-- 统一注册事件
local function RegisterCheckboxEvents(_Parent)
    local _Checkboxes = {"SoundCheckbox", "MusicCheckbox", "VibrationCheckbox"}
    for _, id in ipairs(_Checkboxes) do
        local _Handle = GetWindowHandle(id)
        if _Handle ~= 0 then
            GUI:WndRegistScript(_Handle, RDWndBaseCL_checkbox_change, "OnCheckboxChanged")
        end
    end
end
```

## 常见问题与解决方案

### 1. 复选框不响应点击事件

```lua
-- 检查和修复复选框事件响应问题
local function FixCheckboxEvent(_GUIHandle)
    if _GUIHandle == 0 then
        dbg("无效的控件句柄")
        return
    end

    -- 检查是否可接收消息
    local _CanRevMsg = GUI:WndGetCanRevMsg(_GUIHandle)
    if not _CanRevMsg then
        GUI:WndSetCanRevMsg(_GUIHandle, true)
        dbg("已启用控件消息接收")
    end

    -- 检查是否可用
    local _Enable = GUI:WndGetEnableM(_GUIHandle)
    if not _Enable then
        GUI:WndSetEnableM(_GUIHandle, true)
        dbg("已启用控件")
    end

    -- 检查是否可见
    local _Visible = GUI:WndGetVisibleM(_GUIHandle)
    if not _Visible then
        GUI:WndSetVisibleM(_GUIHandle, true)
        dbg("已显示控件")
    end

    -- 检查选中状态是否可改变
    local _AutoChange = GUI:CheckBoxGetAutoChange(_GUIHandle)
    if not _AutoChange then
        GUI:CheckBoxSetAutoChange(_GUIHandle, true)
        dbg("已启用选中状态改变")
    end
end
```

### 2. 复选框状态不同步

```lua
-- 同步复选框状态
local function SyncCheckboxState(_GUIHandle, _DesiredState)
    if _GUIHandle == 0 then
        return false
    end

    local _CurrentState = GUI:CheckBoxGetCheck(_GUIHandle)

    if _CurrentState ~= _DesiredState then
        GUI:CheckBoxSetCheck(_GUIHandle, _DesiredState)
        dbg(string.format("复选框状态已同步为: %s", tostring(_DesiredState)))
        return true
    end

    dbg(string.format("复选框状态已经是: %s", tostring(_DesiredState)))
    return false
end
```
