# 神途游戏客户端接口学习总结 - 滑动条控件接口

## 概述

滑动条控件是神途游戏客户端中用于提供数值调节功能的基础UI组件，它允许用户通过拖动滑块来选择特定的数值。滑动条通常用于音量控制、亮度调节、进度显示等场景，提供直观的视觉反馈和精确的数值选择。

### 核心功能

- 创建滑动条控件
- 设置和获取当前百分比
- 控制滑块的偏移和位置
- 响应滑动事件
- 自定义滑动条外观

### 应用场景

- 音量调节
- 画质设置
- 技能释放进度
- 任务完成进度
- 装备强化进度

## 基础操作

### 1. 创建滑动条控件

```lua
-- 创建滑动条控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local _GUIHandle = GUI:SlideCreate(_Parent, "ExampleSlide", 1854000012, 1854000013, 1854000014, 100, 100, 612, 23)

if _GUIHandle ~= 0 then
    dbg("滑动条控件创建成功")
    -- 设置默认位置
    GUI:SlideSetPercent(_GUIHandle, 50) -- 默认在50%位置
else
    dbg("滑动条控件创建失败")
end
```

### 2. 设置和获取百分比

```lua
-- 设置滑动条百分比
local _GUIHandle = GetWindowHandle("ExampleSlide")
if _GUIHandle ~= 0 then
    GUI:SlideSetPercent(_GUIHandle, 75) -- 设置为75%
end

-- 获取滑动条百分比
local _Percent = GUI:SlideGetPercent(_GUIHandle)
dbg("滑动条当前位置: " .. _Percent .. "%")
```

### 3. 设置滑块偏移


## 事件处理

### 1. 滑动事件响应

```lua
-- 注册滑动事件
local _GUIHandle = GetWindowHandle("ExampleSlide")
GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_slide_change, "OnSlideChanged")

-- 滑动事件处理函数
function OnSlideChanged(_GUIHandle)
    local _Percent = GUI:SlideGetPercent(_GUIHandle)
    dbg("滑动条位置改变: " .. _Percent .. "%")

    -- 根据百分比执行相应操作
    if _Percent > 80 then
        dbg("音量已设置为高")
    elseif _Percent > 50 then
        dbg("音量已设置为中")
    else
        dbg("音量已设置为低")
    end
end
```

### 2. 滑块点击事件

```lua
-- 注册滑块点击事件
local _GUIHandle = GetWindowHandle("ExampleSlide")
GUI:WndRegistScript(_GUIHandle, RDWndBaseCL_mouse_lbClick, "OnSlideClicked")

-- 点击事件处理函数
function OnSlideClicked(_GUIHandle)
    local _X, _Y = GetMousePos()
    -- 计算点击位置对应的百分比
    local _SlideRect = GetWindowRect(_GUIHandle)
    local _ClickPercent = math.floor((( _X - _SlideRect.left) / _SlideRect.width) * 100)
    _ClickPercent = math.max(0, math.min(100, _ClickPercent))

    -- 设置滑动条位置
    GUI:SlideSetPercent(_GUIHandle, _ClickPercent)
    dbg("点击位置对应百分比: " .. _ClickPercent .. "%")
end
```

```lua
local _MinPercent = 20
```

### 2. 平滑滑动效果

```lua
-- 平滑滑动效果
local function SmoothSlideTo(_GUIHandle, _TargetPercent, _Duration)
    local _CurrentPercent = GUI:SlideGetPercent(_GUIHandle)
    local _Step = (_TargetPercent - _CurrentPercent) / (_Duration / 50) -- 每50ms更新一次
    local _CurrentTime = 0

    -- 创建定时器
    local _TimerId = GUI:WndAddTimer(_GUIHandle, 1001, 50, -1, "OnSlideTimer")

    -- 存储目标值
    GUI:WndSetParam(_GUIHandle, "TargetPercent", _TargetPercent)
    GUI:WndSetParam(_GUIHandle, "Step", _Step)
    GUI:WndSetParam(_GUIHandle, "CurrentTime", _CurrentTime)
    GUI:WndSetParam(_GUIHandle, "TimerId", _TimerId)
end

-- 定时器回调函数
function OnSlideTimer(_GUIHandle)
    local _TimerId = GUI:WndGetParam(_GUIHandle, "TimerId")
    local _CurrentTime = GUI:WndGetParam(_GUIHandle, "CurrentTime") + 50
    local _TargetPercent = GUI:WndGetParam(_GUIHandle, "TargetPercent")
    local _Step = GUI:WndGetParam(_GUIHandle, "Step")
    local _Duration = 1000 -- 1秒

    local _CurrentPercent = GUI:SlideGetPercent(_GUIHandle)
    _CurrentPercent = _CurrentPercent + _Step

    if (_Step > 0 and _CurrentPercent >= _TargetPercent) or (_Step < 0 and _CurrentPercent <= _TargetPercent) then
        GUI:SlideSetPercent(_GUIHandle, _TargetPercent)
        GUI:WndDelTimer(_GUIHandle, _TimerId)
        dbg("平滑滑动完成")
    else
        GUI:SlideSetPercent(_GUIHandle, _CurrentPercent)
        GUI:WndSetParam(_GUIHandle, "CurrentTime", _CurrentTime)
    end
end

-- 使用示例
local _GUIHandle = GetWindowHandle("ExampleSlide")
SmoothSlideTo(_GUIHandle, 70, 1000) -- 1秒内平滑滑动到70%
```

## 常见问题与解决方案

### 1. 滑动条无法响应事件

```lua
-- 检查和修复滑动条事件响应问题
local function FixSlideEvent(_GUIHandle)
    if _GUIHandle == 0 then
        dbg("无效的控件句柄")
        return
    end

    -- 检查是否可接收消息
    local _CanRevMsg = GUI:WndGetCanRevMsg(_GUIHandle)
    if not _CanRevMsg then
        GUI:WndSetCanRevMsg(_GUIHandle, true)
        dbg("已启用控件消息接收")
    end

    -- 检查是否可用
    local _Enable = GUI:WndGetEnableM(_GUIHandle)
    if not _Enable then
        GUI:WndSetEnableM(_GUIHandle, true)
        dbg("已启用控件")
    end

    -- 检查是否可见
    local _Visible = GUI:WndGetVisibleM(_GUIHandle)
    if not _Visible then
        GUI:WndSetVisibleM(_GUIHandle, true)
        dbg("已显示控件")
    end
end
```

### 2. 滑动条位置不准确

```lua
-- 校准滑动条位置
local function CalibrateSlidePosition(_GUIHandle, _ExpectedPercent)
    if _GUIHandle == 0 then
        return false
    end

    local _CurrentPercent = GUI:SlideGetPercent(_GUIHandle)

    if math.abs(_CurrentPercent - _ExpectedPercent) > 1 then
        GUI:SlideSetPercent(_GUIHandle, _ExpectedPercent)
        dbg(string.format("滑动条位置已校准为: %d%%", _ExpectedPercent))
        return true
    end

    dbg(string.format("滑动条位置已在目标范围内: %d%%", _CurrentPercent))
    return false
end

-- 使用校准函数
local _GUIHandle = GetWindowHandle("ExampleSlide")
CalibrateSlidePosition(_GUIHandle, 50)
```
