### 二、编辑框基础属性设置

#### 1. 创建编辑框

```lua
-- 获取父窗口句柄（通常在窗口创建时获取）
local _Parent = LuaGlobal["AttachPartent"]

-- 创建编辑框控件
local _GUIHandle = GUI:EditCreate(
    _Parent,          -- 父窗口句柄
    "Edit0",          -- 控件名称
    160,              -- X坐标
    200,              -- Y坐标
    500,              -- 宽度
    30                -- 高度
)

-- 检查创建是否成功
if _GUIHandle ~= 0 then
    dbg("编辑框创建成功，句柄：" .. _GUIHandle)
else
    dbg("编辑框创建失败")
end
```

#### 2. 设置文本内容

```lua
-- 设置普通文本
GUI:EditSetTextM(_GUIHandle, "这是编辑框的内容")

-- 设置数值文本
GUI:EditSetInt(_GUIHandle, 12345)

-- 获取文本内容
local text = GUI:EditGetTextM(_GUIHandle)
dbg("编辑框文本：" .. text)
```

#### 3. 设置文本颜色

```lua
-- 设置文本颜色（ARGB格式）
GUI:EditSetTextColor(_GUIHandle, GUI:MakeARGB(255, 255, 0, 0))  -- 红色文本

-- 设置文本描边颜色
GUI:EditSetTextGrowColor(_GUIHandle, GUI:MakeARGB(255, 0, 0, 0))  -- 黑色描边
```

### 三、编辑框高级功能

#### 1. 数字编辑框

```lua
-- 设置为数字编辑框
GUI:EditSetBNumber(_GUIHandle, true)

-- 设置最大数值

-- 设置最大字符数

-- 检查是否为数字编辑框
local isNumberEdit = GUI:EditGetBNumber(_GUIHandle)
dbg("是否为数字编辑框：" .. tostring(isNumberEdit))
```

#### 2. 密码编辑框

```lua
-- 设置为密码编辑框
GUI:EditSetIsPassWord(_GUIHandle, true)

-- 设置密码显示模式（仅手游和H5支持）
GUI:EditSetShowPassword(_GUIHandle, true)  -- 明文显示

-- 获取密码模式状态
local isPassword = GUI:EditGetIsPassWord(_GUIHandle)
dbg("是否为密码编辑框：" .. tostring(isPassword))
```

#### 3. 货币编辑框

```lua
-- 设置为货币编辑框

-- 设置货币类型编辑框的最大数值

-- 获取编辑框类型
local maxNumber = GUI:EditGetMaxNumber(_GUIHandle)
dbg("最大数值限制：" .. maxNumber)
```

```lua

GUI:EditSetFontM(_GUIHandle, "SIMLI18")  -- 设置为隶书字体

-- 设置字体大小
GUI:WndSetFontSize(_GUIHandle, 16)

-- 设置文本居中
GUI:EditSetFontCenter(_GUIHandle)

local gradient = {

}
GUI:EditSetGradient(_GUIHandle, gradient)
```

### 五、编辑框事件处理

#### 1. 文本变化事件

```lua
-- 注册文本变化事件
GUI:WndRegistScript(_GUIHandle, "OnTextChange", "OnEditTextChange")

-- 事件处理函数
function OnEditTextChange(wnd, handle)
    local text = GUI:EditGetTextM(handle)
    dbg("编辑框文本变化：" .. text)

    -- 检查文本长度
    local textLength = string.len(text)
    if textLength > 10 then
        GUI:SysMsg_SendWarnMsg("文本过长，请输入少于10个字符")
    end
end
```

#### 2. 编辑状态事件

```lua
-- 注册编辑状态变化事件
GUI:WndRegistScript(_GUIHandle, "OnEditStateChange", "OnEditStateChange")

-- 事件处理函数
function OnEditStateChange(wnd, handle, isEditing)
    dbg("编辑状态变化：" .. tostring(isEditing))
    if isEditing then
        dbg("开始编辑")
    else
        dbg("结束编辑")
    end
end
```

### 七、编辑框常见使用场景

#### 1. 角色名称输入

```lua
-- 创建角色名称输入框
local nameEdit = GUI:EditCreate(_Parent, "NameEdit", 200, 100, 200, 30)
if nameEdit ~= 0 then
    GUI:EditSetTextM(nameEdit, "")
    GUI:EditSetMaxCharNum(nameEdit, 10)
    GUI:EditSetFontM(nameEdit, "TitleFont1")
    GUI:WndSetTextColorM(nameEdit, GUI:MakeARGB(255, 51, 51, 51))
end
```

#### 2. 密码验证

```lua
-- 创建密码输入框
local passwordEdit = GUI:EditCreate(_Parent, "PasswordEdit", 200, 150, 200, 30)
if passwordEdit ~= 0 then
    GUI:EditSetIsPassWord(passwordEdit, true)
    GUI:EditSetTextM(passwordEdit, "")
    GUI:EditSetMaxCharNum(passwordEdit, 20)
end
```

#### 3. 数值输入

```lua
-- 创建数值输入框
local numberEdit = GUI:EditCreate(_Parent, "NumberEdit", 200, 200, 200, 30)
if numberEdit ~= 0 then
    GUI:EditSetBNumber(numberEdit, true)
    GUI:EditSetMaxNumber(numberEdit, 0, "1000")
    GUI:EditSetTextM(numberEdit, "0")
end
```

### 九、编辑框最佳实践

#### 1. 性能优化

```lua
-- 避免频繁创建和销毁编辑框
-- 可以在窗口初始化时创建编辑框，窗口关闭时隐藏而不销毁

-- 限制最大字符数
GUI:EditSetMaxCharNum(_GUIHandle, 20)

-- 避免在文本变化事件中做复杂计算
function OnEditTextChange(wnd, handle)
    local text = GUI:EditGetTextM(handle)
    if string.len(text) > 10 then
        -- 简单提示，避免复杂逻辑
        GUI:SysMsg_SendWarnMsg("文本过长")
    end
end
```

#### 2. 错误处理

```lua
-- 检查编辑框是否创建成功
local _GUIHandle = GUI:EditCreate(_Parent, "Edit0", 160, 200, 500, 30)
if _GUIHandle ~= 0 then
    -- 初始化编辑框属性
    GUI:EditSetTextM(_GUIHandle, "默认值")
else
    dbg("编辑框创建失败")
end
```
