# 神途游戏客户端接口学习总结 - 菜单控件接口

## 概述

菜单控件是神途游戏客户端中用于实现功能分组和选项选择的重要UI组件，它允许将相关功能的按钮组织成一个弹出式菜单，提供清晰的功能分类和直观的交互方式。菜单控件常用于游戏中的右键菜单、功能菜单等场景。

### 核心功能

- 创建菜单控件
- 添加和删除菜单条目
- 设置菜单条目可见性
- 显示和隐藏菜单
- 刷新和重置菜单
- 菜单事件处理

### 应用场景

- 游戏角色右键菜单
- 物品右键菜单
- 地图标记菜单
- 功能菜单分组
- 快捷操作菜单

## 基础操作

### 1. 创建菜单控件

```lua
-- 创建菜单控件
local _Parent = LuaGlobal["AttachPartent"] -- 获取父窗口句柄
local _GUIHandle = GUI:WndCreateWnd(_Parent, "MenuTestWnd", 0, 200, 200)

if _GUIHandle ~= 0 then
    local _MenuHandle = GUI:MenuCreate(_GUIHandle, "MenuId") -- 创建菜单
    Temporary = _MenuHandle -- 通过全局变量传递菜单句柄

    if _MenuHandle ~= 0 then
        -- 添加菜单条目
        GUI:MenuAddItem(0, "测试菜单条目0", 0, -1, 1900340224) -- 添加菜单条目0
        GUI:MenuAddItem(1, "测试菜单条目1", 0, -1, 1900340224) -- 添加菜单条目1

        -- 显示菜单
        GUI:MenuShowPopupMenu(true) -- 在鼠标位置显示菜单

        dbg("菜单创建和显示成功")
    else
        dbg("菜单创建失败")
    end
else
    dbg("父窗口创建失败")
end
```

local function HideMenu()
    local _MenuHandle = GetMenuHandle()

    if _MenuHandle ~= 0 then
        -- 注意：不可在菜单打开状态下删除，否则会导致程序崩溃
        GUI:MenuDeleteMenu()
        Temporary = 0 -- 重置全局变量
        dbg("菜单隐藏成功")
    end
end

-- 定时隐藏菜单
local function AutoHideMenu()
    CL:AddDelayTask("OnAutoHideMenu()", 3000, 1) -- 3秒后隐藏菜单

end

## 优化与性能考虑

### 1. 菜单资源管理

```lua
-- 优化菜单创建和删除
local function OptimizeMenuManagement()
    -- 预创建常用菜单，避免频繁创建和删除
    local _CommonMenus = {}

    -- 创建通用菜单
    local function CreateCommonMenu(_MenuId, _MenuData)
        if not _CommonMenus[_MenuId] then
            _CommonMenus[_MenuId] = CreateMenuFromData(_MenuData)
            dbg("通用菜单" .. _MenuId .. "创建成功")
        end

        return _CommonMenus[_MenuId]
    end

    -- 显示通用菜单
    local function ShowCommonMenu(_MenuId)
        if _CommonMenus[_MenuId] then
            -- 重置菜单状态
            GUI:MenuDestroyMenu()
            -- 重新添加菜单条目
            for _, _MenuItem in ipairs(_MenuData) do
                GUI:MenuAddItem(_MenuItem.ItemId, _MenuItem.Text, _MenuItem.FlagImage, _MenuItem.ParentId, _MenuItem.ItemImg)
            end

            GUI:MenuShowPopupMenu(true)
            return true
        else
            dbg("菜单" .. _MenuId .. "不存在")
            return false
        end
    end

    -- 清理菜单
    local function CleanupMenus()
        for _MenuId, _MenuHandle in pairs(_CommonMenus) do
            GUI:MenuDeleteMenu()
            dbg("菜单" .. _MenuId .. "已清理")
        end

        _CommonMenus = {}
    end

    return {
        CreateCommonMenu = CreateCommonMenu,
        ShowCommonMenu = ShowCommonMenu,
        CleanupMenus = CleanupMenus
    }
end
```
