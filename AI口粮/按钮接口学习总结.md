## 神途客户端按钮接口学习总结

### 一、按钮控件概述

按钮控件是神途客户端中最常用的交互控件，允许用户通过点击操作执行特定功能。按钮支持丰富的外观自定义和交互效果，是构建游戏界面的核心组件之一。

**平台兼容性**：
- ✅ 2D手游：全面支持
- ✅ 3D手游：全面支持
- ✅ 2D端游：全面支持
- ✅ 3D端游：全面支持
- ❌ H5：部分支持

### 二、按钮创建与基础操作

#### 1. 创建按钮控件

```lua
-- 获取父窗口句柄（通常在窗口加载回调中获取）
local _Parent = LuaGlobal["AttachPartent"]

-- 创建按钮控件
local btn_handle = GUI:ButtonCreate(
    _Parent,        -- 父窗口句柄
    "MyButton",     -- 按钮名称（用于识别）
    4010210000,     -- 按钮资源ID
    100,            -- X坐标（相对于父窗口）
    100             -- Y坐标（相对于父窗口）
)

-- 检查按钮创建是否成功
if btn_handle ~= 0 then
    dbg("按钮创建成功，句柄为: " .. btn_handle)
else
    dbg("按钮创建失败")
end
```

**参数说明**：
- `_Parent`：父窗口句柄，按钮必须依附于窗口
- `_Name`：按钮名称，用于后续操作和识别
- `_ResourceID`：按钮资源ID，用于加载按钮图像
- `_X, _Y`：按钮在父窗口内的坐标

#### 2. 基础属性设置

```lua
-- 设置按钮文字
GUI:WndSetTextM(btn_handle, "点击我")

-- 设置按钮大小
GUI:WndSetSizeM(btn_handle, 120, 40)

-- 设置按钮位置（相对于父窗口）
GUI:WndSetPosM(btn_handle, 150, 150)

-- 设置文字颜色（ARGB格式，白色）
GUI:WndSetTextColorM(btn_handle, 4294967295)

-- 设置文字大小（端游和3D手游支持）
GUI:WndSetFontSize(btn_handle, 14)

-- 获取按钮属性
local btn_text = GUI:WndGetText(btn_handle)
local btn_size = GUI:WndGetSizeM(btn_handle)
local btn_pos = GUI:WndGetPosM(btn_handle)

dbg("按钮文字: " .. btn_text)
dbg(string.format("按钮大小: %dx%d", btn_size[1], btn_size[2]))
dbg(string.format("按钮位置: (%d, %d)", btn_pos[1], btn_pos[2]))
```

### 六、按钮事件处理

#### 1. 注册点击事件

```lua
-- 在窗口加载完成后注册事件
function OnWndLoad(wnd_handle)
    -- 获取按钮句柄
    local btn_handle = GUI:WndFindChildM(wnd_handle, "MyButton")

    -- 注册左键点击事件
    GUI:WndRegistScript(btn_handle, "OnLButtonUp", "OnMyButtonClick")

    -- 注册右键点击事件
    GUI:WndRegistScript(btn_handle, "OnRButtonUp", "OnMyButtonRightClick")
end

-- 左键点击事件处理
function OnMyButtonClick(wnd_handle, btn_handle)
    dbg("按钮被左键点击")

    -- 执行点击后的逻辑
    CL:SysMsg_SendPromptMsg("按钮点击成功！")

    -- 暂时禁用按钮
    GUI:WndSetEnableM(btn_handle, false)
    GUI:ButtonSetShowDisable(btn_handle, true)

    -- 2秒后恢复按钮
    CL:AddTimer(btn_handle, 1001, 2000, 1, "OnButtonEnable")
end

-- 右键点击事件处理
function OnMyButtonRightClick(wnd_handle, btn_handle)
    dbg("按钮被右键点击")
    CL:SysMsg_SendPromptMsg("右键功能待实现")
end

-- 按钮恢复启用事件
function OnButtonEnable(wnd_handle, timer_id)
    local btn_handle = wnd_handle  -- 计时器ID与按钮句柄关联
    GUI:WndSetEnableM(btn_handle, true)
    GUI:ButtonSetShowDisable(btn_handle, false)
    dbg("按钮已恢复启用")
end
```

### 七、按钮控件最佳实践

#### 1. 按钮资源管理

```lua
-- 推荐的按钮资源结构
local BUTTON_RESOURCES = {
    -- 普通按钮
    COMMON = 4010210000,
    -- 确定按钮
    CONFIRM = 4010210001,
    -- 取消按钮
    CANCEL = 4010210002,
    -- 红色警告按钮
    WARNING = 4010210003
}

-- 创建标准按钮函数
function CreateStandardButton(parent_handle, name, x, y, text, resource_id)
    local btn = GUI:ButtonCreate(parent_handle, name, resource_id or BUTTON_RESOURCES.COMMON, x, y)
    if btn ~= 0 then
        GUI:WndSetTextM(btn, text)
        GUI:WndSetSizeM(btn, 100, 35)
        GUI:WndSetEnableM(btn, true)
        GUI:ButtonSetDownTextPosChange(btn, true)
    end
    return btn
end
```

#### 2. 按钮状态管理

```lua
-- 按钮状态管理器
local ButtonState = {
    ENABLED = 0,
    DISABLED = 1,
    LOADING = 2,
    SELECTED = 3
}

function SetButtonState(btn_handle, state)
    local btn = btn_handle

    if state == ButtonState.ENABLED then
        GUI:WndSetEnableM(btn, true)
        GUI:ButtonSetShowDisable(btn, false)
    elseif state == ButtonState.DISABLED then
        GUI:WndSetEnableM(btn, false)
        GUI:ButtonSetShowDisable(btn, true)
    elseif state == ButtonState.LOADING then
        GUI:WndSetEnableM(btn, false)
        GUI:ButtonSetShowDisable(btn, false)
        -- 可以添加加载动画效果
    elseif state == ButtonState.SELECTED then
        GUI:ButtonSetIsActiveBtn(btn, true)
        GUI:ButtonSetShine(btn, true)
    end
end

-- 使用示例
local my_btn = CreateStandardButton(_Parent, "ActionBtn", 100, 100, "开始")
SetButtonState(my_btn, ButtonState.DISABLED)  -- 禁用按钮

-- 5秒后启用
CL:AddTimer(my_btn, 1001, 5000, 1, "OnEnableButton")
function OnEnableButton(btn_handle)
    SetButtonState(btn_handle, ButtonState.ENABLED)
end
```

### 八、常见问题与解决方案

#### 1. 按钮不响应点击

```lua
-- 排查点击不响应问题
function DebugButtonClick(btn_handle)
    -- 检查按钮是否启用
    local is_enabled = GUI:WndGetEnableM(btn_handle)
    dbg("按钮是否启用: " .. (is_enabled and "是" or "否"))

    -- 检查按钮是否可见
    local is_visible = GUI:WndGetVisible(btn_handle)
    dbg("按钮是否可见: " .. (is_visible and "是" or "否"))

    -- 检查点击事件是否注册
    local script_info = GUI:WndGetRegistScript(btn_handle, "OnLButtonUp")
    dbg("点击事件脚本: " .. (script_info or "未注册"))

    -- 检查按钮是否被其他控件遮挡
    local is_top = GUI:WndGetIsTopM(btn_handle)
    dbg("按钮是否在最上层: " .. (is_top and "是" or "否"))
end
```

#### 2. 按钮文字被截断

```lua
-- 解决文字截断问题
function AdjustButtonSizeForText(btn_handle, text, padding)
    -- 估算文字宽度（简单方法）
    local char_count = string.len(text)
    local text_width = char_count * 10  -- 每个汉字约10像素
    local btn_width = text_width + (padding or 20)  -- 添加内边距

    -- 设置新尺寸
    GUI:WndSetSizeM(btn_handle, btn_width, GUI:WndGetSizeM(btn_handle)[2])

    -- 重新设置文字
    GUI:WndSetTextM(btn_handle, text)

    dbg("按钮宽度调整为: " .. btn_width)
end

-- 使用示例
AdjustButtonSizeForText(my_btn, "非常长的按钮文字", 30)
```

#### 3. 按钮闪烁问题

```lua
-- 解决闪烁问题
function FixButtonFlicker(btn_handle)
    -- 禁用硬件加速（如支持）
    if GUI:IsSupportHardwareAcceleration() then
        GUI:DisableHardwareAcceleration(btn_handle)
    end

    -- 减少闪烁频率
    GUI:ButtonSetShineGradient(btn_handle, 5)

    -- 优化绘制方式
    GUI:ButtonSetDrawCenter(btn_handle, true)
end
```
